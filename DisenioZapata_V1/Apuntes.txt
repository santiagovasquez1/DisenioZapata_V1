           
           **Calculo de cortante ultimo en dos direcciones**
           float d, A1, A2, Ac, Vu;

            d = Zapata.H - Zapata.R;
            A1 = Zapata.L2 * ((Zapata.L1 - d - Zapata.LcX)/2);
            A2 = (d - Zapata.LcX) * ((Zapata.L2 - Zapata.LcY - d) / 2);
            Ac = A1 + A2;
            Vu = 1.4f * Qmax * Ac;
            return Vu;
